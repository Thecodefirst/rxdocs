                                  # 基础用法

                                  适用于多个勾选框绑定到同一个数组的情景，通过是否勾选来表示这一组选项中选中的项。

                                  TIP

                                  支持RxCheckboxOptions配置(推荐)、template模板插槽两种模式

                                  TIP

                                  RxCheckboxOptions配置，默认开启文字溢出，可通过设置toolTip属性为false关闭文字溢出功能

                                  基础 demo
                                  <template>
  <el-card shadow="never">
    <!-- template 模板插槽 -->
    <rx-checkbox-group
      :checkBoxOptions="checkboxOptions"
      v-model="checked"
    >
      <el-row :gutter="10">
        <el-col :span="4" v-for="(item, index) in checkboxOptions" :key="index" >
          <rx-checkbox :label="item.label"></rx-checkbox>
        </el-col>
      </el-row>
    </rx-checkbox-group>
    <!-- RxCheckboxOptions 配置 -->
    <rx-checkbox-group
      :gutter="10"
      :baseColProps="{ span: 4 }"
      :checkBoxOptions="checkboxOptions"
      v-model="checked"
    ></rx-checkbox-group>
  </el-card>
</template>
<script setup lang="ts">
import { ref } from 'vue'
const checked = ref([])
const checkboxOptions = ref([
  {
    value: false,
    label: 'checkbox1',
  },
  {
    value: false,
    label: 'checkbox2'
  },
  {
    value: false,
    label: 'checkbox3'
  }
])
</script>

                                  通过Row、Col控制RxCheckbox布局

                                  TIP

                                  baseColProps通配布局结构，checkboxOptions.colProps单独配置会覆盖baseColProps

                                  RxCheckbox布局 demo
                                  // baseColProps 类型参数
export interface ColProps {
  tag?: string
  span?: number
  offset?: number
  pull?: number
  push?: number
  xs?: number | ColObject
  sm?: number | ColObject
  md?: number | ColObject
  lg?: number | ColObject
  xl?: number | ColObject
}

baseColProps: {
  type: Object as PropType<ColProps>,
  default: () =>
    ({
      xl: 4,
      lg: 6,
      md: 8,
      sm: 12,
      xs: 24
    } as ColProps)
}

                                  # 文字溢出

                                  toolTip 属性开启且默认为true，设定 toolTip 值为 false，即可禁用该功能

                                  TIP

                                  溢出滚动高度默认80px，可通过toolTipConfig.maxHeight修改溢出滚动高度
                                  默认继承父容器宽度，RxToolTip 宽度自动计算与 RxCheckbox 容器等宽，通过设置toolTipConfig.tooltipWidth可自定义toolTip宽度

                                  WARNING

                                  开启toolTip，必须配置checkBoxOptions属性

                                  文字溢出 demo
                                  interface ToolTipScrollbarCustomType {
  maxHeight?: number
  tooltipWidth?: string
}

interface RxTooltipPropsType = ElTooltip

toolTipConfig: {
  type: Object as PropType<RxTooltipPropsType & ToolTipScrollbarCustomType>,
  default: {
    placement: 'top'
  }
}

                                  # RxCheckboxGroup 插槽

                                  RxCheckbox函数式插槽

                                  TIP: RxCheckbox

                                  1. 使用checkBoxOptions.slots函数式插槽(推荐)，单独配置对应RxCheckbox 默认插槽
                                  RxCheckboxGroup & RxCheckbox 插槽 demo
                                  <template>
  <el-card shadow="never">
     <!-- checkBoxOptions.slots -->
    <rx-checkbox-group
      :baseColProps="{ span: 2 }"
      :checkBoxOptions="checkboxOptions"
      :RxCheckboxAll="true"
      v-model="checkedFormatter"
    ></rx-checkbox-group>
     <!-- 默认插槽 -->
    <rx-checkbox-group
      :baseColProps="{ span: 2 }"
      :checkBoxOptions="checkboxOptions"
      :RxCheckboxAll="true"
      v-model="checkedDefault"
    >
      <rx-checkbox
        v-for="(item, index) in checkboxOptions"
        :key="index"
        :label="item.label"
        size="large"
      ></rx-checkbox>
    </rx-checkbox-group>
  </el-card>
</template>
<script setup lang="ts">
import { ref, h } from 'vue'
const checkedFormatter = ref([])
const checkedDefault = ref([])
const checkboxOptions = ref([
  {
    value: false,
    label: 'checkbox1',
    toolTipContent: 'content-checkbox1'
  },
  {
    value: false,
    label: 'checkbox2',
    toolTipContent:
      'content-checkbox2-content-checkbox2-content-checkbox2-content-checkbox2-content-checkbox2-content-checkbox2',
    slots: (value: any) => h('div', {}, `1checkBoxSlots-${value.label}`)
  },
  {
    value: false,
    label: 'checkbox3',
    toolTipContent:
      'content-checkbox2-content-checkbox2-content-checkbox2-content-checkbox2-content-checkbox2-content-checkbox2'
  }
])
</script>

                                  # RxTooltip 插槽

                                  TIP: RxTooltip

                                  1. 使用checkBoxOptions.toolTipSlots函数式插槽(推荐)，单独配置对应RxCheckbox的RxTooltip模板
                                  2. 使用toolTipFormatter函数式插槽(推荐)，通配全量RxTooltip模板
                                  3. 默认default渲染checkBoxOptions.toolTipContent内容，自带滚动

                                  TIP: RxTooltip.content

                                  1. 优先渲染checkBoxOptions.toolTipContent内容，否则默认渲染checkBoxOptions.label内容
                                  RxCheckboxGroup & RxTooltip 插槽 demo
                                  <template>
  <el-card shadow="never">
     <!-- checkBoxOptions.toolTipSlots -->
     <rx-checkbox-group
      :gutter="10"
      :baseColProps="{ span: 4 }"
      :checkBoxOptions="checkboxOptions"
      :RxCheckboxAll="true"
      :toolTipConfig="{
        placement: 'top'
      }"
      v-model="checkedFormatter"
    ></rx-checkbox-group>
    <!-- toolTip插槽 toolTipFormatter -->
    <rx-checkbox-group
      :gutter="10"
      :baseColProps="{ span: 4 }"
      :checkBoxOptions="checkboxOptions"
      :RxCheckboxAll="true"
      :toolTipConfig="{
        placement: 'top'
      }"
      :toolTipFormatter="handleFormatter"
      v-model="checkedFormatter"
    ></rx-checkbox-group>
  </el-card>
</template>
<script setup lang="ts">
import { ref, h } from 'vue'
const checkedFormatter = ref([])
const checkboxOptions = ref([
  {
    value: false,
    label: 'checkbox1-overflow',
  },
  {
    value: false,
    label: 'checkbox2-overflow',
    toolTipContent:
      'content-checkbox2-content-checkbox2-content-checkbox2-content-checkbox2-content-checkbox2-content-checkbox2',
    toolTipSlots: (value: any) => h('div', {}, `toolTipSlots-${value.label}`),
  },
  {
    value: false,
    label: 'checkbox3-overflow',
    toolTipContent:
      'content-checkbox2-content-checkbox2-content-checkbox2-content-checkbox2-content-checkbox2-content-checkbox2'
  }
])
const handleFormatter = (formatter: any) => h('div', {}, `toolTipFormatter-${formatter.label}`)
</script>

                                  # RxCheckAll 插槽

                                  TIP: RxCheckAll

                                  使用template插槽，默认RxCheckbox组件

                                  RxCheckAll 插槽 demo
                                  <template>
  <el-card shadow="never">
    <!-- 默认 -->
    <rx-checkbox-group
      :rxCheckboxAll="true"
      rxCheckboxAllText="默认全选"
      :gutter="10"
      :baseColProps="{ span: 4 }"
      :checkBoxOptions="checkboxOptions"
      v-model="checkedDefalut"
    ></rx-checkbox-group>
    <!-- 函数式插槽 -->
    <rx-checkbox-group
      :rxCheckboxAll="true"
      :gutter="10"
      :baseColProps="{ span: 4 }"
      :checkBoxOptions="checkboxOptions"
      :checkAllSlotsFormatter="handleFormatter"
      v-model="checkedFormatter"
    ></rx-checkbox-group>
    <!-- template 模板插槽 -->
    <rx-checkbox-group
      :rxCheckboxAll="true"
      :gutter="10"
      :baseColProps="{ span: 4 }"
      :checkBoxOptions="checkboxOptions"
      v-model="checkedTemplate"
    >
      <template #checkAllSlots>
        <span>template全选</span>
      </template>
    </rx-checkbox-group>
  </el-card>
</template>
<script setup lang="ts">
import { ref, h } from 'vue'
const checkedDefalut = ref([])
const checkedFormatter = ref([])
const checkedTemplate = ref([])
const checkboxOptions = ref([
  {
    value: false,
    label: 'checkbox1',
  },
  {
    value: false,
    label: 'checkbox2'
  },
  {
    value: false,
    label: 'checkbox3'

])
const handleFormatter = () => h('span', {}, '全选Formatter')

                                  import type {

  offset?: number
}

export interface ColProps {
  tag?: string
  span?: number
  offset?: number
  pull?: number
  push?: number
  xs?: number | ColObject
  sm?: number | ColObject
  md?: number | ColObject
  lg?: number | ColObject
  xl?: number | ColObject
}

export interface CheckBoxType extends RxCheckBoxComponents {

  toolTipSlots?: Function

}

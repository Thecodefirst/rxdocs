                          # RxSelectPlus 默认开启溢出隐藏，当选项过多时，使用下拉菜单展示并选择内容。

                          TIP

                          适用广泛的基础单选 v-model 的值为当前被选中的 rx-option 的 value 属性值

                          # 基础用法

                          # 基于 RxSelect 基础上扩展，可以通过 options 属性配置 RxOption选项

                          基础 demo
                          <template>
  <el-space wrap>
    <rx-select-plus
      ref="rxSelectPlusRef"
      :toolTip="false"
      v-model="value"
      :options="options"
      placeholder="Select"
      size="large"
      style="width: 200px"
    />
    <rx-select-plus
      ref="rxSelectPlusRef"
      :toolTip="false"
      v-model="value"
      :options="options"
      placeholder="Select"
      style="width: 200px"
    />
    <rx-select-plus
      ref="rxSelectPlusRef"
      :toolTip="false"
      v-model="value"
      :options="options"
      placeholder="Select"
      size="small"
      style="width: 200px"
    />
  </el-space>
</template>
<script lang="ts">
import { defineComponent, ref } from 'vue'
import type { RxSelectPlusExpose } from '@rx-compo/components'
export default defineComponent({
  setup() {
    const value = ref('')
    const rxSelectPlusRef = ref<RxSelectPlusExpose>()
    const options = ref([
      {
        value: 'RxOption1',
        label: 'RxOption1'
      },
      {
        value: 'RxOption2',
        label: 'RxOption2'
      },
      {
        value: 'RxOption3',
        label: 'RxOption3'
      },
      {
        value: 'RxOption4',
        label: 'RxOption4'
      },
      {
        value: 'RxOption5',
        label: 'RxOption5'
      }
    ])

    return { value, options, rxSelectPlusRef }
  }
})
</script>

                          # 文字溢出

                          # toolTip 属性开启且默认为true，设定 toolTip 值为 false，即可禁用该功能

                          TIP

                          溢出滚动高度默认80px，可通过toolTipConfig.maxHeight修改溢出滚动高度
                          默认继承父容器宽度，RxToolTip 宽度自动计算与 RxSelectPlus 容器等宽，通过设置toolTipConfig.tooltipWidth可自定义toolTip宽度

                          WARNING

                          开启toolTip，必须配置options属性

                          文字溢出 demo

                          # 插槽

                          # 可以使用 RxSelect 全部插槽

                          TIP

                          插槽优先级：默认插槽 > RxOptions slots > formatter

                          options slots

                          formatter

                          options slots

                          prefix

                          empty

                          插槽 demo
                          <template>
  <el-card shadow="never">
    <el-space direction="vertical">
      <div>
        <p>options slots</p>
        <rx-select-plus
          ref="rxSelectPlusRef"
          v-model="value"
          :options="options"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
        >
          <rx-option v-for="(item, index) in options" :key="index" v-bind:="item" />
        </rx-select-plus>
      </div>
      <div>
        <p>formatter</p>
        <rx-select-plus
          ref="rxSelectPlusRef"
          v-model="value"
          :options="options"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
          :formatter="handleFormatter"
        />
      </div>
      <div>
        <p>options slots</p>
        <rx-select-plus
          ref="rxSelectPlusRef"
          v-model="value"
          :options="options"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
          :formatter="handleFormatter"
        />
      </div>
      <div>
        <p>prefix</p>
        <rx-select-plus
          ref="rxSelectPlusRef"
          :toolTip="true"
          v-model="value"
          :options="options"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
        >
          <rx-option v-for="(item, index) in options" :key="index" v-bind:="item" />
          <template #prefix> prefix </template>
        </rx-select-plus>
      </div>
      <div>
        <p>empty</p>
        <rx-select-plus
          ref="rxSelectPlusRef"
          :toolTip="true"
          :options="[]"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
        >
          <template #empty> <el-empty description="description" /> </template>
        </rx-select-plus>
      </div>
    </el-space>
  </el-card>
</template>
<script lang="ts">
import { defineComponent, ref, h } from 'vue'
import type { RxSelectPlusExpose } from 'rx-compo/es/components'
export default defineComponent({
  setup() {
    const value = ref([])
    const rxSelectPlusRef = ref<RxSelectPlusExpose>()
    const handleFormatter = (formatter) => h('div', {}, `formatter-${formatter.label}`)
    const options = ref([
      {
        value: 'RxOption1',
        label: 'RxOption1',
        slots: (item) => h('div', {}, `formatter-${item.label}`)
      },
      {
        value: 'RxOption2',
        label: 'RxOption2'
      },
      {
        value: 'RxOption3',
        label: 'RxOption3'
      },
      {
        value: 'RxOption4',
        label: 'RxOption4'
      },
      {
        value: 'RxOption5',
        label: 'RxOption5'
      }
    ])

    return { value, options, rxSelectPlusRef, handleFormatter }
  }
})
</script>

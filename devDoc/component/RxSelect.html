                                # 基础多选

                                # 多选选择器使用 tag 组件来展示已选中的选项。

                                # 为RxSelect设置 multiple 属性即可启用多选， 此时 v-model 的值为当前选中值所组成的数组。 默认情况下选中值会以 Tag 组件的形式展现， 你也可以设置 collapse-tags 属性将它们合并为一段文字。 您可以使用 collapse-tags-tooltip 属性来启用鼠标悬停折叠文字以显示具体所选值的行为。

                                default

                                use collapse-tags

                                use collapse-tags-tooltip

                                use max-collapse-tags

                                基础多选 demo
                                <template>
  <el-card shadow="never">
    <el-space direction="vertical">
      <div>
        <p>default</p>
        <rx-select
          multiple
          v-model="value"
          :options="options"
          placeholder="Select"
          style="width: 300px"
        />
      </div>
      <div>
        <p>use collapse-tags</p>
        <rx-select
          multiple
          collapse-tags
          :toolTip="true"
          v-model="value"
          :options="options"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
        />
      </div>
      <div>
        <p>use collapse-tags-tooltip</p>
        <rx-select
          multiple
          collapse-tags
          collapse-tags-tooltip
          :toolTip="true"
          v-model="value"
          :options="options"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
        />
      </div>
      <div>
        <p>use max-collapse-tags</p>
        <rx-select
          multiple
          collapse-tags
          collapse-tags-tooltip
          :max-collapse-tags="3"
          :toolTip="true"
          v-model="value"
          :options="options"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
        />
      </div>
    </el-space>
  </el-card>
</template>

import { defineComponent, ref } from 'vue'
export default defineComponent({
  setup() {
    const value = ref([])
    const options = ref([
      {
        value: 'RxOption1',
        label: 'RxOption1'
      },
      {
        value: 'RxOption2',
        label: 'RxOption2'
      },
      {
        value: 'RxOption3',
        label: 'RxOption3'
      },
      {
        value: 'RxOption4',
        label: 'RxOption4'
      },
      {
        value: 'RxOption5',
        label: 'RxOption5'

    return { value, options }

})

                                # 插槽

                                # 可以使用 RxSelect 默认插槽、formatter、RxOptions slots三种形式

                                TIP

                                插槽优先级：默认插槽 > RxOptions slots > formatter

                                options slots

                                formatter

                                options slots

                                prefix

                                empty

                                插槽 demo
                                <template>
  <el-card shadow="never">
    <el-space direction="vertical">
      <div>
        <p>options slots</p>
        <rx-select
          v-model="value"
          :options="options"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
        >
          <rx-option v-for="(item, index) in options" :key="index" v-bind:="item" />
        </rx-select>
      </div>
      <div>
        <p>formatter</p>
        <rx-select
          v-model="value"
          :options="options"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
          :formatter="handleFormatter"
        />
      </div>
      <div>
        <p>options slots</p>
        <rx-select
          v-model="value"
          :options="options"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
          :formatter="handleFormatter"
        />
      </div>
      <div>
        <p>prefix</p>
        <rx-select
          v-model="value"
          :options="options"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
        >
          <rx-option v-for="(item, index) in options" :key="index" v-bind:="item" />
          <template #prefix> prefix </template>
        </rx-select>
      </div>
      <div>
        <p>empty</p>
        <rx-select
          :options="[]"
          placeholder="Select"
          style="width: 300px"
          :fit-input-width="true"
        >
          <template #empty> <el-empty description="description" /> </template>
        </rx-select>
      </div>
    </el-space>
  </el-card>
</template>
<script lang="ts">
import { defineComponent, ref, h } from 'vue'
export default defineComponent({
  setup() {
    const value = ref([])
    const handleFormatter = (formatter) => h('div', {}, `formatter-${formatter.label}`)
    const options = ref([
      {
        value: 'RxOption1',
        label: 'RxOption1',
        slots: (item) => h('div', {}, `formatter-${item.label}`)
      },
      {
        value: 'RxOption2',
        label: 'RxOption2'
      },
      {
        value: 'RxOption3',
        label: 'RxOption3'
      },
      {
        value: 'RxOption4',
        label: 'RxOption4'
      },
      {
        value: 'RxOption5',
        label: 'RxOption5'
      }
    ])

    return { value, options, handleFormatter }
  }
})
</script>
